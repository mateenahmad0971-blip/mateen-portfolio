class Book:
    def __init__(self, title, author, price):
        self.title = title
        self.author = author
        self.price = price

class Cart:
    def __init__(self):
        self.items = []

    def add_book(self, book):
        self.items.append(book)
        print(f"Added: {book.title}")

    def total(self):
        return sum(b.price for b in self.items)
